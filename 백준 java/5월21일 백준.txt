package first;

import java.util.*;
public class Main{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		int a, b;
		a = sc.nextInt();
		b = sc.nextInt();
		System.out.println(a + b);
		System.out.println(a - b);
		System.out.println(a * b);
		System.out.println(a / b);
		System.out.println(a % b);
	}
}

import java.util.*;
public class Main{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		int a, b,c;
		a = sc.nextInt();
		b = sc.nextInt();
		c = sc.nextInt();
		int x,y,z,t;
		x = (a+b)%c;
		y = ((a%c)+(b%c))%c;
		z = (a*b)%c;
		t = ((a%c)*(b%c))%c;
		System.out.println(x);
		System.out.println(y);
		System.out.println(z);
		System.out.println(t);
	}
}
